```python
from datetime import datetime as dt
from datetime import timedelta as td

def main():
    days_w = ['', 'пн', 'вт', 'ср', 'чт', 'пт', 'сб', 'вс']
    today = dt.today()
    print(
        f"Сегодня {today.date()}. "

    result = today + td(days = n)
    print(
        f"Через {n} дней будет {result.date()}. "
        f"День недели - {days_w[result.isoweekday()]}"
    )

if __name__ == '__main__':
    main()
```

## Самостоятельная работа №1
### Дайте подробный комментарий для кода, написанного ниже. Комментарий нужен для каждой строчки кода, нужно описать что она делает. Не забудь те, что функции комментируются по-особенному.

```python
# импортируем модуль datetime для работы с датой и временем
from datetime import datetime
# импортируем функцию sqrt из модуля math
from math import sqrt

# определяем главную функцию main, **kwargs - параметр, принимающий
# n колво аргументов
def main(**kwargs):
    # цикл for, который проходит по всем элементам словаря kwargs
    # key будет содержать кортежи
    for key in kwargs.items():
        # вычисление квадратного корня из суммы квадратов словаря kwargs
        # key[1][0] - первое число из значения текущего аргумента
        # key[1][1] - первое число из значения текущего аргумента
        # Результат - гипотенуза(сумма квадратов катета)
        result = sqrt(key[1][0] ** 2 + key[1][1] ** 2)
        # Вывод результата
        print(result)

# проверка запущен ли скрипт напрямую(а не импортирован как модуль)
if __name__ == '__main__':
    # запись текущего времени в переменную start_time
    start_time = datetime.now()
    # вызов функции main с передачей именованных аргументов
    # каждый аргумент - список из двух чисел
    main(
        one = [10, 3], # вектор с координатами(10, 3)
        two = [5, 4], # вектор с координатами(5, 4)
        three = [15, 13], # вектор с координатами(15, 13)
        four = [93, 53], # вектор с координатами(93, 53)
        five = [133, 15] # вектор с координатами(133, 15)
    )

# вычисление времени выполнения программы    
time_costs = datetime.now() - start_time
# вывод времени выполнения программы
print(f"Время выполнения программы - {time_costs}")
```

## Самостоятельная работа №3
### Напишите программу, которая будет выводить текущее время, с точностью до секунд на протяжении 5 секунд. Программу нужно написать с использованием цикла. Подсказка: необходимо использовать модуль datetime и time, а также вам необходимо как-то “усыплять” программу на 1 секунду.

```python
import datetime
import time

def display_time():

    counter = 0
    while counter < 5:
        current_time = datetime.datetime.now()
        formatted_time = current_time.strftime("%H:%M:%S")
        print(f"Текущее время: {formatted_time}")
        
        counter += 1
        if counter < 5:
            time.sleep(1)

if __name__ == "__main__":
    display_time()
```
### Результат
![Меню](https://github.com/vomiraq/eduSOFTENG/blob/%D0%A2%D0%B5%D0%BC%D0%B0_4/pic/s3.png)

### Вывод
Программа создает функцию display_time(), которая инициализирует счетчик counter = 0. В цикле while при условии counter < 5 получает текущее системное время через datetime.datetime.now(), преобразует его в строковый формат "Часы:Минуты:Секунды" с помощью strftime("%H:%M:%S") и выводит с префиксом "Текущее время:". После каждого вывода, кроме последнего, программа останавливается на 1 секунду через time.sleep(1), а счетчик увеличивается на 1. ## Самостоятельная работа №4
### Напишите программу, которая считает среднее арифметическое от аргументов вызываемое функции, с условием того, что изначальное количество этих аргументов неизвестно. Программу необходимо реализовать используя одну функцию и “точку входа”.

```python
def calculate_average(*args):
    if len(args) == 0:
        print("Ошибка: не передано ни одного аргумента")
        return None
    total_sum = sum(args)
    count = len(args)
    average = total_sum / count
    
    print(f"Передано аргументов: {count}")
    print(f"Аргументы: {args}")
    print(f"Сумма аргументов: {total_sum}")
    print(f"Среднее арифметическое: {average:.2f}")
    
    return average

if __name__ == "__main__":
    
    print("Пример 1:")
    calculate_average(1, 2, 3, 4, 5)
    
    print("\nПример 2:")
    calculate_average(10, 20, 30)
    
    print("\nПример 3:")
    calculate_average(2.5, 3.7, 1.8, 4.2)
    
    print("\nПример 4 (один аргумент):")
    calculate_average(7)
    
    print("\nПример 5 (без аргументов):")
    calculate_average()
```

```python
import math

def triangle_area(a, b, c):

    area = math.sqrt(p * (p - a) * (p - b) * (p - c))
    return area

from calc import triangle_area
```
