def tuple_sort(tpl):
    for elm in tpl:
        if not isinstance(elm, int):
            return tpl
    return tuple(sorted(tpl))
    
if __name__ == '__main__':

def remove1st(tuple1, elm):
    list1 = list(tuple1)

    if elm in list1:
        list1.remove(elm)
    return tuple(list1)

print(remove1st((1, 2, 3), 1))

### Программа реализует функцию remove1st(tuple1, elm), которая имитирует удаление первого вхождения заданного элемента (elm) из неизменяемого кортежа (tuple1). Для выполнения этой задачи, функция сначала преобразует входной кортеж в список (list1 = list(tuple1)), так как списки являются изменяемыми. Затем она проверяет, существует ли элемент в списке (if elm in list1). Если элемент найден, используется метод list1.remove(elm), который удаляет первое совпадение. Наконец, измененный список преобразуется обратно в кортеж (return tuple(list1)) и возвращается. Если элемент не найден, список не изменяется, и функция возвращает исходный кортеж, что соответствует условию задания.

```python
def top3_digits(s):
    counts = {}
    for ch in s:
        digit = int(ch)
        if digit in counts:
            counts[digit] += 1
        else:
            counts[digit] = 1
    sorted_counts = sorted(counts.items(), key=lambda x: x[1], reverse=True)
    top3 = sorted_counts[:3]
    result = dict(sorted(top3, key=lambda x: x[0]))
    return result

s = "123456789012345"
print(top3_digits(s))

```
![Меню](https://github.com/vomiraq/eduSOFTENG/blob/%D0%A2%D0%B5%D0%BC%D0%B0_6/pic/s3.PNG)
### Программа реализует функцию top3_digits(s), которая сначала итерирует по входной строке чисел, используя словарь counts для подсчета частоты вхождения каждой цифры в виде целого числа. Затем пары (цифра, частота) из словаря сортируются по убыванию частоты, и из этого списка выбираются три самых частых элемента (top3). Наконец, эти три пары сортируются по возрастанию ключа (самой цифры) и преобразуются обратно в словарь, который и возвращается, успешно находя и форматируя три наиболее повторяющиеся цифры.

def poluchit_chast_kortezha(kortezh, element):
    if element not in kortezh:
        return ()
    perviy = kortezh.index(element)
    try:
        vtoroy = kortezh.index(element, perviy + 1)
        return kortezh[perviy:vtoroy + 1]
    except ValueError:
        return kortezh[perviy:]

print(poluchit_chast_kortezha((1, 2, 3), 8))            
print(poluchit_chast_kortezha((1, 8, 3, 4, 8, 8, 9, 2), 8)) 
print(poluchit_chast_kortezha((1, 2, 8, 5, 1, 2, 9), 8))

Если элемент встречается только один раз, возвращается срез, идущий до конца кортежа. Если элемент не найден вовсе, функция возвращает пустой кортеж (). Для определения границ среза используются метод index() с указанием начальной позиции, а для обработки случая единственного вхождения применяется блок try-except для перехвата ValueError.

def otbor_studentov(spisok_studentov, porog):
    rezultaty = []
    for student in spisok_studentov:
        imya, ocenki = student
        srednee = sum(ocenki) / len(ocenki)
        if srednee >= porog:
            rezultaty.append((imya, srednee))
    rezultaty.sort(key=lambda x: x[1], reverse=True)
    itog = [imya for imya, _ in rezultaty]
    return itog
